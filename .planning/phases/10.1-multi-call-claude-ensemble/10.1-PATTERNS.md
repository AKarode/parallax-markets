```python
    pricing = _PRICING.get(model, _PRICING["sonnet"])
    cost = (input_tokens / 1000) * pricing["input"] + (output_tokens / 1000) * pricing["output"]
    self._spend_today += cost
    self._call_count += 1
```

```python
class MockUsage:
    input_tokens: int = 500
    output_tokens: int = 200

@dataclass
class MockContentBlock:
    text: str

@dataclass
class MockResponse:
    content: list[MockContentBlock]
    usage: MockUsage

def make_mock_client(response_json: dict):
    mock = MagicMock()
    mock_response = MockResponse(
        content=[MockContentBlock(text=json.dumps(response_json))],
        usage=MockUsage(),
    )
    async def mock_create(**kwargs):
        return mock_response
    mock.messages.create = mock_create
    return mock
```
