    **scripts/parallax-cron.sh** -- Wrapper for all cron invocations:
    ```
    #!/usr/bin/env bash
    set -euo pipefail
    source /tmp/parallax-env.sh
    TIMESTAMP=$(date +%Y-%m-%d-%H%M)
    LOG_DIR="$HOME/parallax-logs"
    ERROR_DIR="$LOG_DIR/errors"
    mkdir -p "$LOG_DIR" "$ERROR_DIR" "$LOG_DIR/runs"
    RUN_LOG="$LOG_DIR/$TIMESTAMP.log"
    CMD="$@"
    echo "[$TIMESTAMP] Running: $CMD" >> "$RUN_LOG"
    SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
    PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
    cd "$PROJECT_DIR/backend"
    if python -m parallax.cli.brief $CMD >> "$RUN_LOG" 2>&1; then
        echo "[$TIMESTAMP] SUCCESS" >> "$RUN_LOG"
    else
        EXIT_CODE=$?
        echo "[$TIMESTAMP] FAILED (exit $EXIT_CODE)" >> "$RUN_LOG"
        echo "{\"timestamp\": \"$TIMESTAMP\", \"command\": \"$CMD\", \"exit_code\": $EXIT_CODE}" > "$ERROR_DIR/$TIMESTAMP.json"
    fi
    ```

    **scripts/cron-health-check.sh** -- Nightly summary of successes/failures (D-04):
    ```
    #!/usr/bin/env bash
    ERROR_DIR="$HOME/parallax-logs/errors"
    TODAY=$(date +%Y-%m-%d)
    echo "=== Parallax Health Check: $TODAY ==="
    ERRORS=$(find "$ERROR_DIR" -name "${TODAY}*.json" 2>/dev/null | wc -l | tr -d ' ')
    LOGS=$(find "$HOME/parallax-logs" -maxdepth 1 -name "${TODAY}*.log" 2>/dev/null | wc -l | tr -d ' ')
    echo "Runs today: $LOGS"
    echo "Failures: $ERRORS"
    if [ "$ERRORS" -gt 0 ]; then
        echo ""
        echo "--- Error Details ---"
        cat "$ERROR_DIR"/${TODAY}*.json 2>/dev/null
    fi
    echo "=== End Health Check ==="
    ```
